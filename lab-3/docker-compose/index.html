services:

    ports:
      - "8093:8001"

      DB_HOST: db
      DB_NAME: finance_db
      DB_USER: postgres
      DB_PASSWORD: 1234
    depends_on:
      - db
    networks:
      - my_network

  db:
    image: postgres:14
